    console.warn('⚠️ 암호학적으로 안전한 난수 생성기를 사용할 수 없습니다. Math.random()을 사용합니다.');
    for (let i = 0; i < randomBytes.length; i++) {

  // 타임스탬프 추가로 고유성 보장
  const timestamp = Date.now().toString(16);

    .join('');
